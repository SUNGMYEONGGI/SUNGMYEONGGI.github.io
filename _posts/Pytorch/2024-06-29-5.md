# RNN
> CLASS torch.nn.RNN(input_size, hidden_size, num_layers=1, nonlinearity='tanh', bias=True, batch_first=False, dropout=0.0, bidirectional=False, device=None, dtype=None)


입력 시퀀스에 다층 Elman RNN을 tanh 또는 ReLU 비선형성을 적용하십시오. 입력 시퀀스의 각 요소에 대해 각 계층은 다음 함수를 계산합니다:

$$
h_t = \tanh(x_t W_{ih}^T + b_{ih} + h_{t-1} W_{hh}^T + b_{hh})
$$

여기서 $$h_t$$는 시간 $$t$$에서의 은닉 상태이고, $$x_t$$는 시간 $$t$$에서의 입력이며, $$h(t-1)$$는 시간 $$t-1$$에서의 이전 계층의 은닉 상태 또는 시간 0에서의 초기 은닉 상태입니다. 비선형성이 `relu`인 경우에는 tanh 대신 ReLU가 사용됩니다.
```python
# Efficient implementation equivalent to the following with bidirectional=False
def forward(x, h_0=None):
    if batch_first:
        x = x.transpose(0, 1)
    seq_len, batch_size, _ = x.size()
    if h_0 is None:
        h_0 = torch.zeros(num_layers, batch_size, hidden_size)
    h_t_minus_1 = h_0
    h_t = h_0
    output = []
    for t in range(seq_len):
        for layer in range(num_layers):
            h_t[layer] = torch.tanh(
                x[t] @ weight_ih[layer].T
                + bias_ih[layer]
                + h_t_minus_1[layer] @ weight_hh[layer].T
                + bias_hh[layer]
            )
        output.append(h_t[-1])
        h_t_minus_1 = h_t
    output = torch.stack(output)
    if batch_first:
        output = output.transpose(0, 1)
    return output, h_t
```

## Parameters
- input_size: 입력 `x`의 특성 수입니다.
- hidden_size: 은닉 상태 `h`의 특성 수입니다.
- num_layers: 순환 계층의 개수입니다. 예를 들어, `num_layers=2`로 설정하면 두 개의 RNN을 쌓아서 스택된 RNN을 형성합니다. 두 번째 RNN은 첫 번째 RNN의 출력을 입력으로 받아 최종 결과를 계산합니다. 기본값은 1입니다.
- nonlinearity: 사용할 비선형성입니다. 'tanh' 또는 'relu' 중 하나를 선택할 수 있습니다. 기본값은 'tanh'입니다.
- bias: False인 경우, RNN 계층은 편향 가중치 $$b_{ih}$$와 $$b_{hh}$$를 사용하지 않습니다. 기본값은 True입니다.
- batch_first: True인 경우, 입력 및 출력 텐서는 (배치, 시퀀스, 특성) 형식으로 제공됩니다. 기본값은 False이며, 이는 은닉 상태나 셀 상태에는 적용되지 않습니다.
- dropout: 0이 아닌 경우, 각 RNN 계층의 출력에 Dropout 레이어를 추가합니다. 마지막 계층을 제외한 모든 계층에 대해 Dropout 확률이 지정됩니다. 기본값은 0입니다.
- bidirectional: True인 경우, 양방향 RNN이 됩니다. 기본값은 False입니다.

## PyTorch Code
```python
import torch
import torch.nn as nn
import torch.optim as optim
from torch.utils.data import DataLoader, TensorDataset
from torchvision import transforms
from torch.nn.utils.rnn import pad_sequence
import matplotlib.pyplot as plt
from keras.datasets import imdb

# IMDB 데이터셋을 로드합니다
(x_train, y_train), (x_test, y_test) = imdb.load_data(num_words=10000)

# 모든 시퀀스를 같은 길이로 패딩합니다
max_len = 500
x_train = pad_sequence([torch.tensor(seq) for seq in x_train], batch_first=True, padding_value=0)[:, :max_len]
x_test = pad_sequence([torch.tensor(seq) for seq in x_test], batch_first=True, padding_value=0)[:, :max_len]
x_train = x_train[:, :max_len]
x_test = x_test[:, :max_len]

y_train = torch.tensor(y_train)
y_test = torch.tensor(y_test)

# 데이터 로더를 만듭니다
batch_size = 32
train_data = TensorDataset(x_train, y_train)
test_data = TensorDataset(x_test, y_test)
train_loader = DataLoader(train_data, shuffle=True, batch_size=batch_size)
test_loader = DataLoader(test_data, shuffle=False, batch_size=batch_size)

# 모델을 정의합니다
class RNNModel(nn.Module):
    def __init__(self, vocab_size, embed_size, hidden_size):
        super(RNNModel, self).__init__()
        self.embedding = nn.Embedding(vocab_size, embed_size)
        self.rnn = nn.RNN(embed_size, hidden_size, batch_first=True, dropout=0.2)
        self.fc = nn.Linear(hidden_size, 1)
        self.sigmoid = nn.Sigmoid()
    
    def forward(self, x):
        x = self.embedding(x)
        x, _ = self.rnn(x)
        x = self.fc(x[:, -1, :])
        return self.sigmoid(x)

# 하이퍼파라미터 설정
vocab_size = 10000
embed_size = 128
hidden_size = 128

# 모델 초기화
model = RNNModel(vocab_size, embed_size, hidden_size)
criterion = nn.BCELoss()
optimizer = optim.Adam(model.parameters(), lr=0.001)

# 모델을 학습합니다
num_epochs = 10
train_losses = []
val_losses = []
train_accuracies = []
val_accuracies = []

for epoch in range(num_epochs):
    model.train()
    train_loss = 0
    train_correct = 0
    for x_batch, y_batch in train_loader:
        optimizer.zero_grad()
        output = model(x_batch)
        loss = criterion(output.squeeze(), y_batch.float())
        loss.backward()
        optimizer.step()
        
        train_loss += loss.item()
        predicted = (output.squeeze() >= 0.5).float()
        train_correct += (predicted == y_batch).sum().item()
    
    train_loss /= len(train_loader)
    train_accuracy = train_correct / len(train_loader.dataset)
    train_losses.append(train_loss)
    train_accuracies.append(train_accuracy)
    
    model.eval()
    val_loss = 0
    val_correct = 0
    with torch.no_grad():
        for x_batch, y_batch in test_loader:
            output = model(x_batch)
            loss = criterion(output.squeeze(), y_batch.float())
            val_loss += loss.item()
            predicted = (output.squeeze() >= 0.5).float()
            val_correct += (predicted == y_batch).sum().item()
    
    val_loss /= len(test_loader)
    val_accuracy = val_correct / len(test_loader.dataset)
    val_losses.append(val_loss)
    val_accuracies.append(val_accuracy)
    
    print(f'Epoch {epoch+1}/{num_epochs}, Train Loss: {train_loss:.4f}, Train Accuracy: {train_accuracy:.4f}, Val Loss: {val_loss:.4f}, Val Accuracy: {val_accuracy:.4f}')

# 손실 값과 정확도 결과를 플롯합니다
plt.plot(train_losses, label='Train Loss')
plt.plot(val_losses, label='Validation Loss')
plt.title('Model Loss')
plt.ylabel('Loss')
plt.xlabel('Epoch')
plt.legend(loc='upper right')
plt.show()

plt.plot(train_accuracies, label='Train Accuracy')
plt.plot(val_accuracies, label='Validation Accuracy')
plt.title('Model Accuracy')
plt.ylabel('Accuracy')
plt.xlabel('Epoch')
plt.legend(loc='lower right')
plt.show()
```
